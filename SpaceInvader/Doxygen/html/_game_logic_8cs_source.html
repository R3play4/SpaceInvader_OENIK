244  public void UfoMoveSideways()
245  {
246  if (!this.IsSideMovingUfoDisplayed())
247  {
248  // 20% chance that a sidemoving UFO will appear
249  if (this.r.Next(1, 100) > 90)
250  {
251  // 50% chance of moving left or right.
252  if (this.r.Next(1, 100) > 50)
253  {
254  Ufo sideMovingUFO = new Ufo(20, 25, 100);
255  this.ufoMovingRight = true;
256  this.Model.Ufos.Add(sideMovingUFO);
257  }
258  else
259  {
260  Ufo sideMovingUFO = new Ufo((int)(Settings.WindowWidth - 20), 25, 100);
261  this.ufoMovingRight = false;
262  this.Model.Ufos.Add(sideMovingUFO);
263  }
264  }
265  }
266  else
267  {
268  // There is already a moving UFO. Loops through all the UFO's and moves the one with 100 points sideways.
269  for (int i = 0; i < this.Model.Ufos.Count(); i++)
270  {
271  Ufo actualUFO = this.Model.Ufos[i];
272 
273  if (actualUFO.Points == 100 && this.ufoMovingRight == true)
274  {
275  actualUFO.MoveSideWays(Settings.UfoSideStepSize);
276  }
277  else if (actualUFO.Points == 100 && this.ufoMovingRight == false)
278  {
279  actualUFO.MoveSideWays(-Settings.UfoSideStepSize);
280  }
281 
282  if (actualUFO.X < 0 || actualUFO.X > Settings.WindowWidth)
283  {
284  this.Model.Ufos.Remove(actualUFO);
285  }
286  }
287  }
288  }
